## Usage

### Single import

```tsx
import { TextField } from "~/components/ui/text-field";
```

```tsx
<TextField.Root>
  <TextField.Input placeholder="Email" />
</TextField.Root>
```

### Multiple imports

```tsx
import { TextFieldRoot, TextFieldInput } from "~/components/ui/text-field";
```

```tsx
<TextFieldRoot>
  <TextFieldInput placeholder="Email" />
</TextFieldRoot>
```
