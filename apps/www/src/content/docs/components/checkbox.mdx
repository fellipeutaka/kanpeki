## Usage

### Single import

```tsx
import { Checkbox } from "~/components/ui/checkbox";
```

```tsx
<Checkbox.Root>
  <Checkbox.Indicator />
</Checkbox.Root>
```

### Multiple imports

```tsx
import { CheckboxRoot, CheckboxIndicator } from "~/components/ui/checkbox";
```

```tsx
<CheckboxRoot>
  <CheckboxIndicator />
</CheckboxRoot>
```
